5fn parse_ticket_node_info(ticket: &str) -> Result<(), Box<dyn std::error::Error>> {
6    // Remove "ticket:" prefix
7    let encoded = ticket.strip_prefix("ticket:")
8        .ok_or("Invalid ticket format")?;
9
10    // Decode base32
11    let ticket_json_bytes = BASE32.decode(encoded.as_bytes())?;
12    let ticket_json = String::from_utf8(ticket_json_bytes)?;
13
14    // Parse JSON
15    let ticket_data: serde_json::Value = serde_json::from_str(&ticket_json)?;
16
17    // Extract node information
18    let node_id = ticket_data.get("node_id")
19        .and_then(|v| v.as_str())
20        .ok_or("Missing node_id in ticket")?;
21
22    let endpoint_addr_b64 = ticket_data.get("endpoint_addr")
23        .and_then(|v| v.as_str())
24        .ok_or("Missing endpoint_addr in ticket")?;
25
26    let created_at = ticket_data.get("created_at")
27        .and_then(|v| v.as_u64())
28        .ok_or("Missing created_at in ticket")?;
29
30    println!("Node ID: {}", node_id);
31    println!("Endpoint Addr (base64): {}", endpoint_addr_b64);
32    println!("Created At: {}", created_at);
33
34    // Now test the SerializableEndpointAddr parsing
35    test_serializable_endpoint_addr(endpoint_addr_b64)?;
36
37    Ok(())
38}

40fn test_serializable_endpoint_addr(b64_str: &str) -> Result<(), Box<dyn std::error::Error>> {
41    println!("Testing SerializableEndpointAddr parsing...");
42    println!("Input string length: {}", b64_str.len());
43    println!("Input string: {}", b64_str);
44
45    // Check for invalid characters (similar to our Rust implementation)
46    let valid_chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=";
47    let invalid_chars: Vec<char> = b64_str.chars()
48        .filter(|c| !valid_chars.contains(*c))
49        .collect();
50
51    if !invalid_chars.is_empty() {
52        println!("❌ Found invalid characters: {:?}", invalid_chars);
53        return Err(format!("Invalid characters found: {:?}", invalid_chars).into());
54    }
55
56    // Check length
57    if b64_str.len() % 4 != 0 {
58        println!("❌ Length is not a multiple of 4: {}", b64_str.len());
59        return Err(format!("Length not multiple of 4: {}", b64_str.len()).into());
60    }
61
62    // Try base64 decode (using new API)
63    let engine = base64::engine::general_purpose::STANDARD;
64
65    match engine.decode(b64_str) {
66        Ok(decoded) => {
67            let decoded: Vec<u8> = decoded;
68            println!("✅ Base64 decode successful, {} bytes", decoded.len());
69
70            let json_str = String::from_utf8(decoded)?;
71            println!("✅ UTF-8 decode successful");
72            println!("JSON: {}", json_str);
73
74            let endpoint_data: serde_json::Value = serde_json::from_str(&json_str)?;
75            println!("✅ JSON parse successful");
76            println!("Endpoint data: {}", serde_json::to_string_pretty(&endpoint_data)?);
77
78            Ok(())
79        }
80        Err(e) => {
81            println!("❌ Base64 decode failed: {}", e);
82            Err(format!("Base64 decode failed: {}", e).into())
83        }
84    }
85}
